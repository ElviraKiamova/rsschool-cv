    function returnsNumber(str) {
      let string = '';
      if (typeof str === 'number') {
        string = parseInt(Math.abs(str).toString().replace('.', ''), 10);
      }
      for(let i = 0; i <= str.length - 1; i++) {
        const pars = Number.isNaN(parseInt(str[i], 10));
        if (pars === false) {
          string += str[i];
        }
      }
      if (string === '') {
        return NaN;
      }
      return +string;
    }
